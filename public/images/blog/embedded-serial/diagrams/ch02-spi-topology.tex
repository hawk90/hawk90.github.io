\draw[bus] (bus2) -- (bus2 |- s1.west) -- (s1.west |- bus2);
\draw[bus] (bus2) -- (bus2 |- s3.west) -- (s3.west |- bus2);

\draw[bus] (bus3) -- (bus3 |- s3.west) -- (s3.west |- bus3);

\draw[cs] (m.south) ++(2mm, 0) -- ++(0, -6mm) -| ([xshift=-3mm]s2.south) -- ++(0, 3mm) node[lbl, midway, left] {CS2};
\draw[cs] (m.south) ++(4mm, 0) -- ++(0, -9mm) -| ([xshift=-3mm]s3.south) -- ++(0, 3mm) node[lbl, midway, left] {CS3};
