  {LP-01/00\\[1pt]\tiny entry};
\node[font=\scriptsize\bfseries, anchor=north, align=center] at (5.25, 6.4)
  {HS payload\\[1pt]\tiny DDR $\pm$200\,mV};
\node[font=\scriptsize\bfseries, anchor=north, align=center] at (8.0, 6.4)
  {Trail};
\node[font=\scriptsize\bfseries, anchor=north, align=center] at (9.1, 6.4)
  {LP-11};

% Dp signal (y around 3.0-4.5)
\node[lbl] at (-0.1, 3.5) {Dp};

% LP-11: high (4.3)
\draw[lp] (0.0, 4.3) -- (1.2, 4.3);
% LP-01: Dp low (3.3)
\draw[lp] (1.2, 4.3) -- (1.2, 3.3) -- (2.1, 3.3);
% LP-00: stays low
\draw[lp] (2.1, 3.3) -- (3.0, 3.3);

% HS Sync + payload: differential around 3.7 (mid)
% Toggle around mid with ±0.3
\draw[hs] (3.0, 3.3) -- (3.0, 4.0);
\foreach \i in {0,1,...,7} {
  \pgfmathsetmacro{\xa}{3.0 + \i*0.5625}
  \pgfmathsetmacro{\xb}{3.0 + (\i+1)*0.5625}
  \pgfmathtruncatemacro{\level}{mod(\i,2)}
  \ifnum\level=0
    \draw[hs] (\xa, 4.0) -- (\xb, 4.0);
    \ifnum\i<7
      \draw[hs] (\xb, 4.0) -- (\xb, 3.4);
    \fi
  \else
    \draw[hs] (\xa, 3.4) -- (\xb, 3.4);
    \ifnum\i<7
      \draw[hs] (\xb, 3.4) -- (\xb, 4.0);
    \fi
  \fi
}
% Trail
\draw[hs] (7.5, 3.4) -- (7.5, 3.7);
\draw[lp] (7.5, 3.7) -- (8.5, 3.7);
% Back to LP-11
\draw[lp] (8.5, 3.7) -- (8.5, 4.3) -- (9.7, 4.3);

% Dn signal (y around 0.5-2.0)
\node[lbl] at (-0.1, 1.5) {Dn};

% LP-11: high (1.8)
\draw[lp] (0.0, 1.8) -- (1.2, 1.8);
% LP-01: Dn stays high
\draw[lp] (1.2, 1.8) -- (2.1, 1.8);
% LP-00: Dn low
\draw[lp] (2.1, 1.8) -- (2.1, 0.8) -- (3.0, 0.8);

% HS payload: inverse of Dp around mid 1.3
\draw[hs] (3.0, 0.8) -- (3.0, 1.6);
\foreach \i in {0,1,...,7} {
  \pgfmathsetmacro{\xa}{3.0 + \i*0.5625}
  \pgfmathsetmacro{\xb}{3.0 + (\i+1)*0.5625}
  \pgfmathtruncatemacro{\level}{mod(\i,2)}
  \ifnum\level=0
    \draw[hs] (\xa, 1.0) -- (\xb, 1.0);
    \ifnum\i<7
      \draw[hs] (\xb, 1.0) -- (\xb, 1.6);
    \fi
  \else
    \draw[hs] (\xa, 1.6) -- (\xb, 1.6);
    \ifnum\i<7
      \draw[hs] (\xb, 1.6) -- (\xb, 1.0);
    \fi
  \fi
}
% Trail
\draw[hs] (7.5, 1.0) -- (7.5, 1.3);
\draw[lp] (7.5, 1.3) -- (8.5, 1.3);
% Back to LP-11
\draw[lp] (8.5, 1.3) -- (8.5, 1.8) -- (9.7, 1.8);
