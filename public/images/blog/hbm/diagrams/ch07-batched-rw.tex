  \pgfmathsetmacro\xx{\i*1.5}
  \ifnum\pdfstrcmp{\type}{R}=0
    \node[cls, fill=rdfill, draw=rdbord, minimum width=8mm, minimum height=6mm, anchor=west] at (\xx, 1.8) {\type};
  \else
    \node[cls, fill=wrfill, draw=wrbord, minimum width=8mm, minimum height=6mm, anchor=west] at (\xx, 1.8) {\type};
  \fi
  % turnaround penalty between each (except last)
  \ifnum\i<7
    \node[cls, fill=taFill, draw=taBord, minimum width=5mm, minimum height=4mm, anchor=west, font=\tiny] at (\xx + 0.8, 1.8) {ta};
  \fi
}

% ===== Batched: RRRR then WWWW =====
\node[font=\sffamily\bfseries, anchor=west, text=okbord] at (0,0.5) {Batched — turnaround 최소화};

  \node[cls, fill=rdfill, draw=rdbord, minimum width=8mm, minimum height=6mm, anchor=west] at (\xx, -0.2) {R};
}
% one turnaround
\node[cls, fill=taFill, draw=taBord, minimum width=8mm, minimum height=5mm, anchor=west, font=\footnotesize] at (3.7, -0.2) {ta};
\foreach \i in {0,1,2,3} {
  \pgfmathsetmacro\xx{4.7 + \i*0.9}
  \node[cls, fill=wrfill, draw=wrbord, minimum width=8mm, minimum height=6mm, anchor=west] at (\xx, -0.2) {W};
}
% second turnaround
\node[cls, fill=taFill, draw=taBord, minimum width=8mm, minimum height=5mm, anchor=west, font=\footnotesize] at (8.4, -0.2) {ta};
\foreach \i in {0,1,2,3} {
  \pgfmathsetmacro\xx{9.4 + \i*0.9}
  \node[cls, fill=rdfill, draw=rdbord, minimum width=8mm, minimum height=6mm, anchor=west] at (\xx, -0.2) {R};
}
