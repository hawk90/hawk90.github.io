\documentclass[border=10pt,tikz]{standalone}
\usepackage{fontspec}
\setmainfont{Apple SD Gothic Neo}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,decorations.pathreplacing}
\input{../../_design.tex}

\begin{document}
\begin{tikzpicture}[blog,
  sda/.style={thick, draw=absbord},
  scl/.style={thick, draw=impbord},
  lbl/.style={font=\sffamily\footnotesize, text=inkc},
  bitlbl/.style={font=\sffamily\scriptsize, text=mutec},
  marker/.style={font=\sffamily\scriptsize\bfseries, text=focusc}
]

% Bit width 0.9. Bit i starts at x = 0.5 + i*0.9 (i = 0..8 for 7 addr + R/W + ACK).
% Bit pattern A6..A0 = 1010001, R/W=1, ACK=0

% SDA: pre-start high
\node[lbl, anchor=east] at (-0.2, 2) {SDA};
\draw[sda] (-0.5, 2.7) -- (0, 2.7);
% START: 1 -> 0
\draw[sda] (0, 2.7) -- (0, 2) -- (0.5, 2);
% A6 = 1
\draw[sda] (0.5, 2) -- (0.5, 2.7) -- (1.4, 2.7);
% A5 = 0
\draw[sda] (1.4, 2.7) -- (1.4, 2) -- (2.3, 2);
% A4 = 1
\draw[sda] (2.3, 2) -- (2.3, 2.7) -- (3.2, 2.7);
% A3 = 0
\draw[sda] (3.2, 2.7) -- (3.2, 2) -- (4.1, 2);
% A2 = 0
\draw[sda] (4.1, 2) -- (5.0, 2);
% A1 = 1
\draw[sda] (5.0, 2) -- (5.0, 2.7) -- (5.9, 2.7);
% A0 = 0
\draw[sda] (5.9, 2.7) -- (5.9, 2) -- (6.8, 2);
% R/W = 1
\draw[sda] (6.8, 2) -- (6.8, 2.7) -- (7.7, 2.7);
% ACK = 0 (slave pulls)
\draw[sda] (7.7, 2.7) -- (7.7, 2) -- (8.6, 2);

% SCL: 9 pulses for 8 bits + ACK
\node[lbl, anchor=east] at (-0.2, 0) {SCL};
% pre-start high
\draw[scl] (-0.5, 0.7) -- (0.5, 0.7);
% Drop to low between bits, pulses at center of each bit
% Bit i: low from 0.5+i*0.9 to 0.5+i*0.9+0.05; high from there to 0.5+i*0.9+0.85; low to next
\foreach \i in {0,1,2,3,4,5,6,7,8} {
  \pgfmathsetmacro{\xa}{0.5 + \i * 0.9 + 0.05}
  \pgfmathsetmacro{\xb}{0.5 + \i * 0.9 + 0.85}
  \draw[scl] (\xa, 0.7) -- (\xa, 0) -- (\xb, 0) -- (\xb, 0.7);
}
% Connect tops between pulses
\draw[scl] (0.5, 0.7) -- (0.55, 0.7);
\foreach \i in {0,1,2,3,4,5,6,7} {
  \pgfmathsetmacro{\xa}{0.5 + \i * 0.9 + 0.85}
  \pgfmathsetmacro{\xb}{0.5 + (\i+1) * 0.9 + 0.05}
  \draw[scl] (\xa, 0.7) -- (\xb, 0.7);
}
\draw[scl] (8.35, 0.7) -- (8.7, 0.7);

% Bit labels above SDA
\node[bitlbl] at (0.95, 3.1) {A6};
\node[bitlbl] at (1.85, 3.1) {A5};
\node[bitlbl] at (2.75, 3.1) {A4};
\node[bitlbl] at (3.65, 3.1) {A3};
\node[bitlbl] at (4.55, 3.1) {A2};
\node[bitlbl] at (5.45, 3.1) {A1};
\node[bitlbl] at (6.35, 3.1) {A0};
\node[bitlbl] at (7.25, 3.1) {R/W};
\node[bitlbl, text=conbord] at (8.15, 3.1) {ACK};

% START marker
\node[marker, anchor=south] at (0, 3.55) {START};
\draw[->, draw=focusc, line width=0.5pt] (0, 3.45) -- (0, 2.85);

% Bracket above 7-bit address
\draw[decorate, decoration={brace, amplitude=4pt, mirror},
      draw=mutec, line width=0.4pt]
  (0.5, 3.4) -- (7.7, 3.4);
\node[bitlbl, anchor=south] at (4.1, 3.45) {7-bit address + R/W};

\end{tikzpicture}
\end{document}
