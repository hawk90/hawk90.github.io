\documentclass[border=10pt,tikz]{standalone}
\usepackage{fontspec}
\setmainfont{Apple SD Gothic Neo}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,decorations.pathreplacing}
\input{../../_design.tex}

\begin{document}
\begin{tikzpicture}[blog,
  sig/.style={thick, draw=absbord},
  ctr/.style={thick, draw=mutec},
  cmp/.style={dashed, draw=focusc, line width=0.4pt},
  lbl/.style={font=\sffamily\footnotesize, text=inkc},
  sublbl/.style={font=\sffamily\scriptsize, text=mutec}
]

% --- LEFT: Edge-aligned ---
% Counter: sawtooth, 3 periods
% Period = 2, height = 1, starting at x=0
\def\Tl{2.0}

\node[lbl, anchor=south] at (\Tl*1.5, 4.4) {Edge-aligned};

% Counter (top)
\node[lbl, anchor=east] at (-0.1, 3.5) {counter};
% Three sawtooth ramps
\foreach \i in {0, 1, 2} {
  \pgfmathsetmacro{\x}{\i*\Tl}
  \draw[ctr] (\x, 2.8) -- (\x + \Tl, 3.8) -- (\x + \Tl, 2.8);
}
% Comparator level (CCR)
\draw[cmp] (0, 3.1) -- (3*\Tl, 3.1);
\node[sublbl, text=focusc, anchor=west] at (3*\Tl + 0.05, 3.1) {CCR};

% PWM output (bottom)
\node[lbl, anchor=east] at (-0.1, 1.5) {PWM};
\foreach \i in {0, 1, 2} {
  \pgfmathsetmacro{\x}{\i*\Tl}
  \draw[sig]
    (\x, 1.2) -- (\x, 1.9) -- (\x + 0.6, 1.9) -- (\x + 0.6, 1.2) -- (\x + \Tl, 1.2);
}

% --- RIGHT: Center-aligned ---
\def\xR{6.5}
\def\Tr{2.0}

\node[lbl, anchor=south] at (\xR + \Tr*1.5, 4.4) {Center-aligned};

\node[lbl, anchor=east] at (\xR - 0.1, 3.5) {counter};
% Triangle up-down, period 2*Tr, two periods
\foreach \i in {0, 1} {
  \pgfmathsetmacro{\x}{\xR + \i*2*\Tr}
  \draw[ctr] (\x, 2.8) -- (\x + \Tr, 3.8) -- (\x + 2*\Tr, 2.8);
}
\draw[cmp] (\xR, 3.1) -- (\xR + 4*\Tr, 3.1);
\node[sublbl, text=focusc, anchor=west] at (\xR + 4*\Tr + 0.05, 3.1) {CCR};

\node[lbl, anchor=east] at (\xR - 0.1, 1.5) {PWM};
% PWM goes high when counter > CCR, symmetric around peak
\foreach \i in {0, 1} {
  \pgfmathsetmacro{\x}{\xR + \i*2*\Tr}
  \draw[sig]
    (\x, 1.2) -- (\x + 0.6, 1.2) -- (\x + 0.6, 1.9)
    -- (\x + 2*\Tr - 0.6, 1.9) -- (\x + 2*\Tr - 0.6, 1.2)
    -- (\x + 2*\Tr, 1.2);
}

% Period brace under each
\draw[decorate, decoration={brace, amplitude=3pt, mirror},
      draw=mutec, line width=0.4pt]
  (0, 0.8) -- (\Tl, 0.8);
\node[sublbl, anchor=north] at (\Tl/2, 0.7) {period};

\draw[decorate, decoration={brace, amplitude=3pt, mirror},
      draw=mutec, line width=0.4pt]
  (\xR, 0.8) -- (\xR + 2*\Tr, 0.8);
\node[sublbl, anchor=north] at (\xR + \Tr, 0.7) {period (up+down)};

\end{tikzpicture}
\end{document}
