\documentclass[border=10pt,tikz]{standalone}
\usepackage{fontspec}
\setmainfont{Apple SD Gothic Neo}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,decorations.pathreplacing}
\input{../../_design.tex}

\begin{document}
\begin{tikzpicture}[blog,
  sig/.style={thick, draw=absbord},
  sigm/.style={thick, draw=impbord},
  sigs/.style={thick, draw=conbord},
  cs/.style={thick, draw=hibord},
  cell/.style={draw=mutec, fill=clifill, line width=0.3pt},
  lbl/.style={font=\sffamily\footnotesize, text=inkc},
  bitlbl/.style={font=\sffamily\scriptsize, text=mutec}
]

% Vertical positions
\def\ycs{4.5}
\def\ysck{3.0}
\def\ymosi{1.5}
\def\ymiso{0.0}

% CS: low through entire transaction
\node[lbl, anchor=east] at (-0.3, \ycs + 0.3) {CS};
\draw[cs]
  (-0.5, \ycs + 0.7) -- (0, \ycs + 0.7)
  -- (0, \ycs) -- (8, \ycs)
  -- (8, \ycs + 0.7) -- (8.5, \ycs + 0.7);

% SCK: 16 clock pulses (2 bytes), idle low
\node[lbl, anchor=east] at (-0.3, \ysck + 0.3) {SCK};
\draw[sig] (-0.5, \ysck) -- (0.4, \ysck);
\foreach \i in {0, ..., 15} {
  \pgfmathsetmacro{\x}{0.5 + \i*0.25}
  \draw[sig]
    (\x, \ysck) -- (\x, \ysck + 0.6)
    -- (\x + 0.12, \ysck + 0.6) -- (\x + 0.12, \ysck);
}
\draw[sig] (0.5 + 16*0.25 - 0.13, \ysck) -- (8.5, \ysck);

% MOSI: byte cells (address A5, then data 01)
\node[lbl, anchor=east] at (-0.3, \ymosi + 0.3) {MOSI};
\draw[cell] (0.5, \ymosi) rectangle (2.5, \ymosi + 0.6);
\node[bitlbl] at (1.5, \ymosi + 0.3) {0xA5 (cmd)};
\draw[cell] (2.6, \ymosi) rectangle (4.5, \ymosi + 0.6);
\node[bitlbl] at (3.55, \ymosi + 0.3) {0x01 (arg)};
% Lines for idle
\draw[sigm] (-0.5, \ymosi + 0.3) -- (0.5, \ymosi + 0.3);
\draw[sigm] (4.5, \ymosi + 0.3) -- (8.5, \ymosi + 0.3);

% MISO: only second byte returns data, first is 0x00 (idle)
\node[lbl, anchor=east] at (-0.3, \ymiso + 0.3) {MISO};
\draw[cell] (0.5, \ymiso) rectangle (2.5, \ymiso + 0.6);
\node[bitlbl] at (1.5, \ymiso + 0.3) {0x00};
\draw[cell] (2.6, \ymiso) rectangle (4.5, \ymiso + 0.6);
\node[bitlbl] at (3.55, \ymiso + 0.3) {0x42 (data)};
\draw[sigs] (-0.5, \ymiso + 0.3) -- (0.5, \ymiso + 0.3);
\draw[sigs] (4.5, \ymiso + 0.3) -- (8.5, \ymiso + 0.3);

% Annotation arrows
\draw[->, draw=focusc, line width=0.4pt]
  (1.5, -0.4) -- (1.5, \ymosi - 0.05);
\node[bitlbl, text=focusc, anchor=north] at (1.5, -0.4) {address byte};

\draw[->, draw=focusc, line width=0.4pt]
  (3.55, -0.4) -- (3.55, \ymiso - 0.05);
\node[bitlbl, text=focusc, anchor=north] at (3.55, -0.4) {data byte};

\end{tikzpicture}
\end{document}
