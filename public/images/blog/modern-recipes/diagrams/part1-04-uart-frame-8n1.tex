\documentclass[border=10pt,tikz]{standalone}
\usepackage{fontspec}
\setmainfont{Apple SD Gothic Neo}
\usepackage{tikz}
\usetikzlibrary{positioning,arrows.meta,decorations.pathreplacing}
\input{../../_design.tex}

\begin{document}
\begin{tikzpicture}[blog,
  sig/.style={thick, draw=absbord},
  lbl/.style={font=\sffamily\footnotesize, text=inkc},
  bitlbl/.style={font=\sffamily\scriptsize, text=mutec},
  region/.style={draw=clibord, fill=clifill, fill opacity=0.4, line width=0.3pt}
]

% Geometry: idle 1.5, start 1, D0..D7 each 1, stop 1, idle 1.5
% Each bit width = 1
\def\bw{0.9}

% Background highlight regions
\fill[region] (1.5*\bw, -0.15) rectangle (2.5*\bw, 1.15);
\fill[region] (10.5*\bw, -0.15) rectangle (11.5*\bw, 1.15);

% Waveform: idle(1) start(0) D0..D7(varying) stop(1) idle(1)
% sequence: 1,1.5  ->  0 at start, then bits 10110100 example, then stop 1, idle 1
% Simpler: show all bits as alternating to make waveform interesting
\draw[sig]
  (0, 1) -- (1.5*\bw, 1)           % idle high
  -- (1.5*\bw, 0) -- (2.5*\bw, 0)  % start bit low
  -- (2.5*\bw, 1) -- (3.5*\bw, 1)  % D0 = 1
  -- (3.5*\bw, 0) -- (4.5*\bw, 0)  % D1 = 0
  -- (4.5*\bw, 1) -- (5.5*\bw, 1)  % D2 = 1
  -- (5.5*\bw, 1) -- (6.5*\bw, 1)  % D3 = 1
  -- (6.5*\bw, 0) -- (7.5*\bw, 0)  % D4 = 0
  -- (7.5*\bw, 1) -- (8.5*\bw, 1)  % D5 = 1
  -- (8.5*\bw, 0) -- (9.5*\bw, 0)  % D6 = 0
  -- (9.5*\bw, 0) -- (10.5*\bw, 0) % D7 = 0
  -- (10.5*\bw, 1) -- (11.5*\bw, 1)% stop bit high
  -- (11.5*\bw, 1) -- (13*\bw, 1); % idle high

% Bit labels above
\node[bitlbl] at (2*\bw,  1.35) {Start};
\foreach \i/\x in {0/3, 1/4, 2/5, 3/6, 4/7, 5/8, 6/9, 7/10} {
  \node[bitlbl] at (\x*\bw - 0.5*\bw, 1.35) {D\i};
}
\node[bitlbl] at (11*\bw, 1.35) {Stop};

% Section labels below
\node[lbl] at (0.75*\bw, -0.55) {idle};
\node[lbl] at (2*\bw,   -0.55) {start};
\node[lbl] at (6.5*\bw, -0.55) {data (LSB first)};
\node[lbl] at (11*\bw,  -0.55) {stop};
\node[lbl] at (12.25*\bw, -0.55) {idle};

% Level annotations
\node[lbl, anchor=east] at (-0.15, 1) {1};
\node[lbl, anchor=east] at (-0.15, 0) {0};

% Falling edge marker
\draw[->, draw=focusc, line width=0.5pt]
  (1.5*\bw, -0.85) -- (1.5*\bw, -0.15);
\node[lbl, text=focusc, anchor=north] at (1.5*\bw, -0.85) {falling edge $\to$ sample};

\end{tikzpicture}
\end{document}
