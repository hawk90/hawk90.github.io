% AHB bus line

\node[traceblock, right=0.2cm of itm] (tpiu) {TPIU};
\node[traceblock, right=0.2cm of tpiu] (etm) {ETM};

% Connections from bus to blocks
\foreach \blk in {cpu, fpb, dwt, itm, tpiu, etm} {
    \draw[conn] ([yshift=0.3cm]\blk.north) -- (\blk.north);
}

% Connection from DAP to bus
\draw[conn] (dap) -- (bustext.north);

% Annotation for trace output
